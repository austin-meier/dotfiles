  (defvar elpaca-installer-version 0.12)
  (defvar elpaca-directory (expand-file-name "elpaca/" user-emacs-directory))
  (defvar elpaca-builds-directory (expand-file-name "builds/" elpaca-directory))
  (defvar elpaca-sources-directory (expand-file-name "sources/" elpaca-directory))
  (defvar elpaca-order '(elpaca :repo "https://github.com/progfolio/elpaca.git"
                                :ref nil :depth 1 :inherit ignore
                                :files (:defaults "elpaca-test.el" (:exclude "extensions"))
                                :build (:not elpaca-activate)))
  (let* ((repo  (expand-file-name "elpaca/" elpaca-sources-directory))
         (build (expand-file-name "elpaca/" elpaca-builds-directory))
         (order (cdr elpaca-order))
         (default-directory repo))
    (add-to-list 'load-path (if (file-exists-p build) build repo))
    (unless (file-exists-p repo)
      (make-directory repo t)
      (when (<= emacs-major-version 28) (require 'subr-x))
      (condition-case-unless-debug err
          (if-let* ((buffer (pop-to-buffer-same-window "*elpaca-bootstrap*"))
                    ((zerop (apply #'call-process `("git" nil ,buffer t "clone"
                                                    ,@(when-let* ((depth (plist-get order :depth)))
                                                        (list (format "--depth=%d" depth) "--no-single-branch"))
                                                    ,(plist-get order :repo) ,repo))))
                    ((zerop (call-process "git" nil buffer t "checkout"
                                          (or (plist-get order :ref) "--"))))
                    (emacs (concat invocation-directory invocation-name))
                    ((zerop (call-process emacs nil buffer nil "-Q" "-L" "." "--batch"
                                          "--eval" "(byte-recompile-directory \".\" 0 'force)")))
                    ((require 'elpaca))
                    ((elpaca-generate-autoloads "elpaca" repo)))
              (progn (message "%s" (buffer-string)) (kill-buffer buffer))
            (error "%s" (with-current-buffer buffer (buffer-string))))
        ((error) (warn "%s" err) (delete-directory repo 'recursive))))
    (unless (require 'elpaca-autoloads nil t)
      (require 'elpaca)
      (elpaca-generate-autoloads "elpaca" repo)
      (let ((load-source-file-function nil)) (load "./elpaca-autoloads"))))
  (add-hook 'after-init-hook #'elpaca-process-queues)
  (when (eq system-type 'windows-nt)
    (elpaca-no-symlink-mode))
  (elpaca `(,@elpaca-order))

    (defun my/eshell-project-or-current-dir ()
      "Open an eshell for the current project or current directory.

         If the current buffer is part of a project (via `project.el`):
         - Try to switch to an existing eshell buffer named after the project root.
         - If it doesn't exist, create a new eshell in the project root.

         If not in a project, open a standard eshell in the current buffer's directory."
      (interactive)
      (let* ((proj (project-current))
             (default-dir (if proj
                              (project-root proj)
                            default-directory))
             (eshell-name (if proj
                              (concat "eshell-" (file-name-nondirectory
                                                 (directory-file-name default-dir)))
                            nil)))
        ;; If project-based eshell exists, switch to it
        (if eshell-name
            (let ((buf (get-buffer eshell-name)))
              (if buf
                  (switch-to-buffer buf)
                ;; Create new eshell in project root with the name
                (let ((default-directory default-dir))
                  (eshell "new")
                  (rename-buffer eshell-name))))
          ;; No project: just open regular eshell in current directory
          (let ((default-directory default-dir))
            (eshell)))))

  ;; --- C/C++ build/run helpers ----------------------------------------------
  ;; C/C++ has no single canonical build command, so these detect the project
  ;; shape (CMake → Makefile → loose file) and `compile' the right thing. Std
  ;; versions are configurable; single-file builds use the portable `cc'/`c++'
  ;; driver names so the same code works with clang (macOS) or gcc (Linux).
  (defvar austin/cc-c-std "c17"
    "C standard passed to single-file compiles.")
  (defvar austin/cc-c++-std "c++20"
    "C++ standard passed to single-file compiles.")
  (defvar austin/cc-cmake-build-dir "build"
    "Out-of-source CMake build directory, relative to the project root.")

  (defun austin/cc--c++-p ()
    "Non-nil when the current buffer is C++ (vs C)."
    (derived-mode-p 'c++-ts-mode 'c++-mode))

  (defun austin/cc--single-file-cmd (&optional run)
    "Compile command for the current buffer; append a run step when RUN."
    (let* ((src (or buffer-file-name (user-error "Buffer is not visiting a file")))
           (out (file-name-sans-extension src)))
      (format "%s -g -O0 -Wall -Wextra -std=%s %s -o %s%s"
              (if (austin/cc--c++-p) "c++" "cc")
              (if (austin/cc--c++-p) austin/cc-c++-std austin/cc-c-std)
              (shell-quote-argument src)
              (shell-quote-argument out)
              (if run (format " && %s" (shell-quote-argument out)) ""))))

  (defun austin/cc--root-with (file)
    "Project root if it contains FILE (a filename), else nil."
    (when-let* ((proj (project-current))
                (root (project-root proj))
                ((file-exists-p (expand-file-name file root))))
      root))

  (defun austin/cc--cmake-root ()
    (austin/cc--root-with "CMakeLists.txt"))

  (defun austin/cc--make-root ()
    (or (austin/cc--root-with "Makefile") (austin/cc--root-with "makefile")))

  (defun austin/cc-build ()
    "Build the current C/C++ project, or compile the current file.
  CMakeLists.txt → configure + build into `austin/cc-cmake-build-dir' (also
  emitting compile_commands.json for clangd); Makefile → run make; otherwise
  compile the current buffer to a binary beside the source."
    (interactive)
    (cond
     ((austin/cc--cmake-root)
      (let* ((root (austin/cc--cmake-root))
             (bdir (expand-file-name austin/cc-cmake-build-dir root)))
        (compile (format "cmake -S %s -B %s -DCMAKE_EXPORT_COMPILE_COMMANDS=ON && cmake --build %s"
                         (shell-quote-argument root)
                         (shell-quote-argument bdir)
                         (shell-quote-argument bdir)))))
     ((austin/cc--make-root)
      (let ((default-directory (austin/cc--make-root)))
        (compile "make")))
     (t (compile (austin/cc--single-file-cmd)))))

  (defun austin/cc-compile-file ()
    "Compile just the current C/C++ buffer to a binary beside the source."
    (interactive)
    (compile (austin/cc--single-file-cmd)))

  (defun austin/cc-run ()
    "Compile the current C/C++ buffer and run the resulting binary."
    (interactive)
    (compile (austin/cc--single-file-cmd t)))

  (defun austin/cc-test ()
    "Run the test suite: ctest in the CMake build dir, else `make test'."
    (interactive)
    (cond
     ((austin/cc--cmake-root)
      (let ((default-directory (expand-file-name austin/cc-cmake-build-dir

        (compile "ctest --output-on-failure")))
     ((austin/cc--make-root)
      (let ((default-directory (austin/cc--make-root)))
        (compile "make test")))

                    nil t))

  ;; --- Ada build/run helpers -------------------------------------------------
  ;; Like C/C++, Ada has several project shapes, so these detect them (Alire →
  ;; GPR → loose file) and `compile' the right thing. `alr' is the Alire package
  ;; manager, `gprbuild' builds an explicit project file, and `gnatmake' builds a
  ;; single source straight to a binary beside it. Reuses `austin/cc--root-with'.
  (defvar austin/ada-gnat-flags "-g -O0 -gnatwa"
    "Flags passed to single-file `gnatmake' builds (debug info, all warnings).")

  (defun austin/ada--alire-root ()
    "Project root if it is an Alire crate (contains alire.toml), else nil."
    (austin/cc--root-with "alire.toml"))

  (defun austin/ada--gpr-file ()
    "Path to the first .gpr project file in the project root, or nil."
    (when-let* ((root (austin/cc--root-with "."))
                (gprs (directory-files root t "\\.gpr\\'")))
      (car gprs)))

  (defun austin/ada--single-file-cmd (&optional run)
    "`gnatmake' command for the current buffer; append a run step when RUN."
    (let* ((src (or buffer-file-name (user-error "Buffer is not visiting a file")))
           (out (file-name-sans-extension src)))
      (format "gnatmake %s %s -o %s%s"
              austin/ada-gnat-flags
              (shell-quote-argument src)
              (shell-quote-argument out)
              (if run (format " && %s" (shell-quote-argument out)) ""))))

  (defun austin/ada-build ()
    "Build the current Ada project, or compile the current file.
  alire.toml → `alr build'; a .gpr file → `gprbuild -P'; otherwise `gnatmake'
  the current buffer to a binary beside the source."
    (interactive)
    (cond
     ((austin/ada--alire-root)
      (let ((default-directory (austin/ada--alire-root)))
        (compile "alr build")))
     ((austin/ada--gpr-file)
      (compile (format "gprbuild -P %s"
                       (shell-quote-argument (austin/ada--gpr-file)))))
     (t (compile (austin/ada--single-file-cmd)))))

** Fonts/Text
#+begin_src emacs-lisp

  ;; Font Style
  ;; Pick a height based on the physical width of the monitor (in mm).
  ;; Larger screen → you're likely sitting further away → bigger font.
  ;; Tune the cond thresholds and heights to taste; these are sensible defaults:
  ;;   laptop / small display  (<380mm wide, e.g. 14–16" MacBook) → 90
  ;;   medium external         (380–599mm,   e.g. 24")             → 110
  ;;   large external          (≥600mm,       e.g. 27"+)           → 130
  (defun austin/font-height-for-display (&optional frame)
    (let* ((attrs (and (display-graphic-p)
                       (frame-monitor-attributes (or frame (selected-frame)))))
           (mm-w  (and attrs (car (alist-get 'mm-size attrs)))))
      (cond ((null mm-w)    90)
            ((>= mm-w 600) 130)
            ((>= mm-w 380) 110)
            (t              90))))

  (defvar font-height (austin/font-height-for-display)
    "Font height chosen at startup from the display's physical size.")
  (setq-default line-spacing 0.2)

  (let ((font-name "AdwaitaMono Nerd Font"))
    (when (member font-name (font-family-list))
      ;; Set the global default first
      (set-face-attribute 'default nil
                          :font font-name
                          :height font-height
                          :width 'expanded
                          :weight 'regular)

      ;; Explicitly ensure the minibuffer prompt matches
      (set-face-attribute 'minibuffer-prompt nil
                          :font font-name
                          :height font-height
                          :weight 'bold) ; Keeping prompt bold for clarity

      ;; If you use a completion UI that uses variable-pitch, 
      ;; force it back to your fixed-width font here:
      (set-face-attribute 'variable-pitch nil
                          :font font-name
                          :height font-height)))

  (use-package nerd-icons-completion
    :config
    (nerd-icons-completion-mode))

  (set-face-attribute 'font-lock-keyword-face nil

  ;; Sane tab defaults

** Prescient (completion sorting)
~orderless~ matches candidates but deliberately does *not* rank them, so
company shows matches in the back end's order — e.g. typing =req= buries
=require= under =custodian-require-memory=, =dynamic-require=, etc.
~prescient~ adds the missing sort: *recency → frequency → length*. Shorter
candidates float up (so =require= leads on length alone), and anything you
pick is remembered and promoted next time. orderless still does the matching;
prescient only reorders. ~prescient-persist-mode~ keeps that history across
sessions.
#+begin_src emacs-lisp
  (use-package prescient
    :config (prescient-persist-mode 1))

  (use-package company-prescient
    :after (company prescient)
    :config (company-prescient-mode 1))

** Debugger
~dap-mode~ drives debugging through the Debug Adapter Protocol; the =SPC c d=
keymap (see Keybindings) is already wired to its commands. For C/C++ we use the

** Electric Pair (auto-close delimiters)
=electric-pair-local-mode= (built-in) auto-inserts matching =()=, =[]=, ={}=,
and =""=, enabled per language in the mode hooks (see
[[#language-specific][Language Specific]]). Structural editing is split: lispyville
gives sexp-safe evil operators in normal/visual state, and paredit's *commands*
handle slurp/barf/wrap/raise in insert state (see Paredit).

We avoid =paredit-mode= itself — its electric open-paren reindents via an
unguarded =backward-up-list=, which =clojure-ts-mode='s tree-sitter syntax
doesn't satisfy (every "(" raised =Scan error: Unbalanced parentheses=). Binding
only paredit's commands, not the mode, sidesteps that entirely.

** Lispyville
Evil-aware structural editing built on lispy. We keep the key theme minimal:
the explicit structural *commands* (slurp/barf, wrap, drag/raise/splice) now
live in INSERT state via paredit (see Paredit / Keybindings → structural
editing), so the only themes enabled here are the ones that aren't redundant
with that:
- ~operators~ — makes evil's =d= / =c= / =y= sexp-safe so normal-state edits
  don't shred unbalanced parens. This is transparent edit *safety*, not a
  command you invoke, which is why it stays.
- ~c-w~ — sexp-aware =C-w= (delete-backward-word) in insert state.

Dropped: ~slurp/barf-cp~, ~wrap~ and ~additional~ (redundant with the insert
bindings), plus ~additional-movement~ (this restores evil's default =H= / =L=
and =[= / =]= motions). Note ~lispyville-mode~ itself stays *on* — the paredit
insert bindings are scoped to ~lispyville-mode-map~.
#+begin_src emacs-lisp
(use-package lispyville
  :hook ((emacs-lisp-mode       . lispyville-mode)
         (lisp-interaction-mode . lispyville-mode)
         (clojure-ts-mode       . lispyville-mode))
  :config
  (lispyville-set-key-theme
   '(operators
     c-w)))
#+end_src

** Paredit (structural-editing commands only)
We deliberately do *not* enable ~paredit-mode~ — its electric open-paren
reindents via an unguarded ~backward-up-list~, which is exactly what raised
=Scan error: Unbalanced parentheses= under ~clojure-ts-mode='s tree-sitter
syntax (see Electric Pair). Instead we pull paredit in only for its
*commands*, bound in INSERT state for lisp buffers (slurp/barf/raise/wrap —
see Keybindings → structural editing). ~:commands~ creates autoloads so
paredit loads lazily on first use, never turning on the mode.

lispyville is kept for what these commands can't give: sexp-safe evil
operators (=d=/=c=/=y=) and structural motions in normal/visual state. The two
are complementary — lispyville owns normal state, paredit owns insert state.
#+begin_src emacs-lisp
  (use-package paredit
    :commands (paredit-forward-slurp-sexp
               paredit-backward-slurp-sexp
               paredit-forward-barf-sexp
               paredit-backward-barf-sexp
               paredit-raise-sexp
               paredit-wrap-round
               paredit-wrap-square
               paredit-wrap-curly))
#+end_src
